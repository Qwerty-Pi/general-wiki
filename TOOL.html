Z3 Solver

Tutorial

from z3 import *

x, y = Ints("x y")
constraints = [x + y == 10, x <= 5, y <= 5]

solver = Solver()

solver.add(constraints)
if solver.check() == sat:
    model = solver.model()
    print(f"Solution: x = {model[x]}, y = {model[y]}")
else:
    print("No solution found.")

OR-Tools

For linear programming. Works in LP better than Z3 Solver. [Source]

from ortools.sat.python import cp_model
model = cp_model.CpModel()

x = [model.new_int_var(-1, 1, f"x{i}") for i in range(3)]
model.add(x[0] * x[1] == x[2])

solver = cp_model.CpSolver()
status = solver.solve(model)
if status == cp_model.OPTIMAL or status == cp_model.FEASIBLE:
    print([solver.value(_x[i]) for i in range(3)])
else:
    print("No solution found.")
